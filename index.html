async function startCam() {
  try {
    cStream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' } });
    document.getElementById('scvid').srcObject = cStream;
    document.getElementById('cam-box').classList.add('open');
    document.getElementById('cam-start').style.display = 'none';
    document.getElementById('cam-stop').style.display = 'block';
    if (typeof ZXing !== 'undefined') {
      cReader = new ZXing.BrowserMultiFormatReader();
      cReader.decodeFromVideoElement(document.getElementById('scvid'), (res) => {
        if (res) {
          document.getElementById('upc-inp').value = res.getText();
          stopCam(); switchScanTab('man'); lookupUPC();
        }
      });
    }
  } catch (e) {
    showToast('Camera not available — use Type UPC tab');
  }
}
function stopCam() {
  if (cReader) { try { cReader.reset(); } catch(e){} cReader = null; }
  if (cStream) { cStream.getTracks().forEach(t => t.stop()); cStream = null; }
  const cb = document.getElementById('cam-box'); if (cb) cb.classList.remove('open');
  const cs = document.getElementById('cam-stop'); if (cs) cs.style.display = 'none';
  const cst = document.getElementById('cam-start'); if (cst) cst.style.display = 'block';
}

function getExpiryDate() {
  let m, y, d = 1;
  if (dateFormat === 'mm') {
    m = parseInt(document.getElementById('dp-month').value);
    y = parseInt(document.getElementById('dp-year').value);
  } else {
    d = parseInt(document.getElementById('dp-day').value) || 1;
    m = parseInt(document.getElementById('dp-month2').value);
    y = parseInt(document.getElementById('dp-year2').value);
  }
  if (!m || !y || m < 1 || m > 12 || y < 2020) return null;
  const day = dateFormat === 'mm' ? new Date(y, m, 0).getDate() : d;
  return new Date(y, m-1, day);
}
function onDateChange() {
  const exp = getExpiryDate();
  const prev = document.getElementById('dp-preview');
  const sug = document.getElementById('dp-suggest');
  if (!exp) { prev.className = 'dp-preview'; sug.className = 'dp-suggest'; return; }
  const today = new Date(); today.setHours(0,0,0,0);
  const diff = Math.floor((exp - today) / 86400000);
  const dateStr = exp.toLocaleDateString('en-CA', { year:'numeric', month:'long', day:'numeric' });
  prev.className = 'dp-preview show';
  prev.innerHTML = `📅 <strong>${dateStr}</strong> · ${diff <= 0 ? '<span style="color:var(--danger)">EXPIRED</span>' : diff + ' day' + (diff!==1?'s':'') + ' remaining'}`;
  let st, col, msg;
  if (diff <= 1)       { st='rm'; col='color:var(--danger);background:rgba(255,71,87,.1);border:1px solid rgba(255,71,87,.3)'; msg='🔴 Expires today or tomorrow — REMOVE FROM SHELF'; }
  else if (diff <= 7)  { st='75'; col='color:#ff6b35;background:rgba(255,107,53,.1);border:1px solid rgba(255,107,53,.3)'; msg='🔶 Expires within 1 week — 75% coupon suggested'; }
  else if (diff <= 14) { st='50'; col='color:var(--org);background:rgba(255,149,0,.1);border:1px solid rgba(255,149,0,.3)'; msg='🟠 Expires within 2 weeks — 50% coupon suggested'; }
  else if (diff <= 30) { st='25'; col='color:var(--w);background:rgba(255,209,102,.1);border:1px solid rgba(255,209,102,.3)'; msg='🟡 Expires within 30 days — 25% coupon suggested'; }
  else                 { st='ok'; col='color:var(--green);background:rgba(6,214,160,.1);border:1px solid rgba(6,214,160,.3)'; msg='✅ Good — no coupon needed yet'; }
  sug.className = 'dp-suggest show';
  sug.style.cssText = col; sug.textContent = msg;
  setSt(st);
}

// ════════════════════════════════════════════════
// ESCALATION ENGINE
// ════════════════════════════════════════════════
function runEscalation() {
  const today = new Date(); today.setHours(0,0,0,0);
  let changed = false;
  logs.forEach(log => {
    if (!log.expiryDate || log.status === 'rm') return;
    const exp = new Date(log.expiryDate);
    const diff = Math.floor((exp - today) / 86400000);
    let newSt;
    if (diff <= 1) newSt = 'rm';
    else if (diff <= 7) newSt = '75';
    else if (diff <= 14) newSt = '50';
    else if (diff <= 30) newSt = '25';
    else newSt = 'ok';
    if (newSt !== log.status) { log.status = newSt; log.escalated = true; changed = true; }
    else log.escalated = false;
  });
  if (changed) saveLogs();
}

function doExport(data, fname) {
  if (!data.length) { showToast('Nothing to export'); return; }
  const hdrs = ['Product','UPC','Planogram ID','Planogram Name','Category','Status','Expiry Date','Units','Notes','Staff','Date Logged','Week'];
  const rows = data.map(l => [
    l.name, l.upc, l.planoId, l.planoName, l.planoCat,
    l.status==='ok'?'OK':l.status==='rm'?'REMOVE':l.status+'% Coupon',
    l.expiryDisplay, l.units, l.notes, l.staff, l.date, 'Week '+l.week
  ].map(v => `"${(v||'').replace(/"/g,'""')}"`).join(','));
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([[hdrs.join(','), ...rows].join('\n')], {type:'text/csv'}));
  a.download = `${fname}-${new Date().toISOString().split('T')[0]}.csv`;
  a.click(); showToast('📥 Exported!');
}
